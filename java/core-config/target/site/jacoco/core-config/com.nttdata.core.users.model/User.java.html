@Setter

	 * @param email {@link String} The email
	 * @return {@link String} the email
	 */
	private String email;
	
	/** 
	 * The profile.
	 * @param profile {@link Profile} The profile
	 * @return {@link Profile} the profile
	 */
	private Profile profile;
}
